9 * @file AST.cpp
10 * @brief AST visitor dispatch and AST export implementations.
11 *
12 * @details
13 * Provides:
14 * - concrete accept() forwarding for double-dispatch,
15 * - text-based AST pretty printer,
16 * - Graphviz DOT exporter with UML-inspired styling,
17 * - file-writing wrappers used by the driver pipeline.

19 * @par Why this organization?
20 * Export/visualization logic is isolated from AST node definitions so traversal
21 * policies can evolve independently from tree structure.

23 * @par What comes next?
24 * Additional exporters (JSON, XML, IR dumps) should follow the same visitor
25 * pattern and be added beside existing ASTPrinter functions.

64/**
65 * @brief Escape DOT label-sensitive characters.
66 * @param text Raw label text.
67 * @return Escaped label safe for Graphviz node/edge labels.
68 */
69std::string escapeDotLabel(const std::string& text) {
70 std::string escaped;
71 escaped.reserve(text.size());
72 for (char c : text) {
73 if (c == '\\' || c == '"' || c == '<' || c == '>') {
74 escaped.push_back('\\');
75 }
76 escaped.push_back(c);
77 }
78 return escaped;
79}

93/**
94 * @class TextASTVisitor
95 * @brief Visitor that serializes AST into a readable indented text tree.
96 *
97 * @details
98 * This representation is optimized for diagnostics and grading output. Sections
99 * like "params", "locals", and "body" make declaration/statement context explicit.
100 */
101class TextASTVisitor : public ASTVisitor {
102 public:
103 std::string str() const { return _out.str(); }
104
105 void visit(IdNode& node) override { visitLeaf(node); }
106 void visit(IntLitNode& node) override { visitLeaf(node); }
107 void visit(FloatLitNode& node) override { visitLeaf(node); }
108 void visit(TypeNode& node) override { visitLeaf(node); }
109 void visit(VarDeclNode& node) override { visitLeaf(node); }
110
111 void visit(BinaryOpNode& node) override { visitBinaryLike(node); }
112 void visit(UnaryOpNode& node) override { visitBinaryLike(node); }
113 void visit(AssignStmtNode& node) override { visitBinaryLike(node); }
114 void visit(WhileStmtNode& node) override { visitBinaryLike(node, "cond", "body"); }
115 void visit(IOStmtNode& node) override { visitBinaryLike(node); }
116 void visit(ReturnStmtNode& node) override { visitBinaryLike(node); }
117
118 void visit(IfStmtNode& node) override {
119 writeNode(node);
120 section("cond");
121 visitChild(node.getLeft(), 0);
122 section("then");
123 visitChild(node.getRight(), 0);
124 if (node.getElseBlock() != nullptr) {
125 section("else");
126 visitChild(node.getElseBlock(), 0);
127 }
128 _depth--;
129 }
130
131 void visit(FuncCallNode& node) override {
132 writeNode(node);
133 if (node.getLeft() != nullptr) {
134 section("callee");
135 visitChild(node.getLeft(), 0);
136 }
137 section("args");
138 for (const auto& arg : node.getArgs()) {
139 visitChild(arg, 1);
140 }
141 _depth--;
142 }
143
144 void visit(DataMemberNode& node) override {
145 writeNode(node);
146 if (node.getLeft() != nullptr) {
147 section("owner");
148 visitChild(node.getLeft(), 0);
149 }
150 if (!node.getIndices().empty()) {
151 section("indices");
152 for (const auto& idx : node.getIndices()) {
153 visitChild(idx, 1);
154 }
155 }
156 _depth--;
157 }
158
159 void visit(BlockNode& node) override {
160 writeNode(node);
161 for (const auto& stmt : node.getStatements()) {
162 visitChild(stmt, 0);
163 }
164 _depth--;
165 }
166
167 void visit(FuncDefNode& node) override {
168 writeNode(node);
169 section("params");
170 for (const auto& param : node.getParams()) {
171 visitChild(param, 1);
172 }
173 section("locals");
174 for (const auto& local : node.getLocalVars()) {
175 visitChild(local, 1);
176 }
177 section("body");
178 visitChild(node.getRight(), 0);
179 _depth--;
180 }
181
182 void visit(ClassDeclNode& node) override {
183 writeNode(node);
184 if (!node.getParents().empty()) {
185 section("inherits");
186 for (const auto& parent : node.getParents()) {
187 line(parent, _depth + 1);
188 }
189 }
190 section("members");
191 for (const auto& member : node.getMembers()) {
192 visitChild(member, 1);
193 }
194 _depth--;
195 }
196
197 void visit(ProgNode& node) override {
198 writeNode(node);
199 section("classes");
200 for (const auto& cls : node.getClasses()) {
201 visitChild(cls, 1);
202 }
203 section("functions");
204 for (const auto& fn : node.getFunctions()) {
205 visitChild(fn, 1);
206 }
207 _depth--;
208 }
209
210 private:
211 std::ostringstream _out;
212 int _depth = 0;
213
214 void line(const std::string& text, int depth) {
215 _out << makeIndent(depth) << text << "\n";
216 }
217
218 void writeNode(ASTNode& node) {
219 line(nodeLabel(node), _depth);
220 _depth++;
221 }
222
223 void section(const std::string& label) {
224 line(label + ":", _depth);
225 }
226
227 void visitChild(const std::shared_ptr<ASTNode>& node, int extraIndent) {
228 if (node == nullptr) {
229 line("<null>", _depth + extraIndent);
230 return;
231 }
232 if (extraIndent == 1) {
233 _depth++;
234 node->accept(*this);
235 _depth--;
236 } else {
237 node->accept(*this);
238 }
239 }
240
241 void visitLeaf(ASTNode& node) {
242 line(nodeLabel(node), _depth);
243 }
244
245 void visitBinaryLike(ASTNode& node, const std::string& leftLabel = "left", const std::string& rightLabel = "right") {
246 writeNode(node);
247 if (node.getLeft() != nullptr) {
248 section(leftLabel);
249 visitChild(node.getLeft(), 0);
250 }
251 if (node.getRight() != nullptr) {
252 section(rightLabel);
253 visitChild(node.getRight(), 0);
254 }
255 _depth--;
256 }
257};

374
375 // UML Inheritance uses an empty arrowhead

410 void declareNode(ASTNode& node, const std::string& shape, const std::string& fillcolor) {
411 const ASTNode* ptr = &node;
412 if (_declared.count(ptr)) return; // Already customized
413 _declared.insert(ptr);
414
415 std::string id = getNodeId(ptr);
416 _nodeLines.push_back(" " + id + " [label=\"" + escapeDotLabel(nodeLabel(node)) + "\", shape=" + shape + ", style=filled, fillcolor=\"" + fillcolor + "\"];");

419 void addEdge(ASTNode& parent, ASTNode& child, const std::string& label, const std::string& arrowhead, const std::string& style) {
420 std::string from = getNodeId(&parent);
421 std::string to = getNodeId(&child);
422 std::string safeLabel = label.empty() ? "relation" : label;
423 _edgeLines.push_back(" " + from + " -> " + to + " [label=\"" + escapeDotLabel(safeLabel) + "\", arrowhead=\"" + arrowhead + "\", style=\"" + style + "\"];");
