1/**
2 * @file semantic.h
3 * @brief Semantic analysis types, symbol tables, and analyzer visitor interface.
4 *
5 * @details
6 * This header defines the semantic layer for the compiler pipeline:
7 * - symbol categories and symbol metadata,
8 * - hierarchical lexical scopes via SymbolTable,
9 * - two-pass semantic analysis over the AST using a visitor.
10 *
11 * @par Why this structure?
12 * Semantic checks require both declaration-time registration (pass 1) and
13 * usage-time validation (pass 2). These APIs keep that separation explicit while
14 * preserving traceable diagnostics with source line numbers.
15 *
16 * @par What comes next?
17 * New language features should extend symbol metadata and analyzer visitors here,
18 * then mirror those changes in parser AST construction and code generation.
19 */

42/**
43 * @struct SymbolEntry
44 * @brief Canonical symbol metadata record stored in symbol tables.
45 *
46 * @details
47 * A single structure is used for classes, functions, variables, parameters, and
48 * fields. Unused members remain empty depending on symbol kind.
49 */
50struct SymbolEntry {
51 /** @brief Symbol identifier. */
52 std::string name;
53 /** @brief Primary type text or signature root. */
54 std::string type;
55 /** @brief Array dimensions for arrays/array parameters (if applicable). */
56 std::vector<int> dimensions;
57 /** @brief Function return type for function symbols. */
58 std::string returnType;
59 /** @brief Function parameter base types. */
60 std::vector<std::string> paramTypes;
61 /** @brief Function parameter dimension specs for array parameters. */
62 std::vector<std::vector<int>> paramDimensions;
63 /** @brief Symbol category. */
64 SymbolKind kind;
65 /** @brief Visibility/role marker (for example local, param, public). */
66 std::string visibility;
67 /** @brief Human-readable semantic details. */
68 std::string details;
69 /** @brief 1-based source declaration line. */
70 int line;
71};

73/**
74 * @class SymbolTable
75 * @brief Hierarchical lexical scope with symbol-definition and resolution services.
76 *
77 * @details
78 * Each scope stores local declarations and links to parent/children scopes.
79 * Resolution first checks current scope then walks upward to parent scopes.
80 */
81class SymbolTable : public std::enable_shared_from_this<SymbolTable> {
82 public:
83 /**
84 * @brief Construct a symbol table scope.
85 * @param scopeName Display name for diagnostics and dumps.
86 * @param parent Parent scope (null for global).
87 */
88 SymbolTable(const std::string& scopeName, std::shared_ptr<SymbolTable> parent = nullptr);
89
90 /** @brief Create and register a child scope below this scope. */
91 std::shared_ptr<SymbolTable> createChild(const std::string& scopeName);
92 /** @brief Define a symbol in current scope; fails on same-scope redeclaration. */
93 bool define(const SymbolEntry& entry);
94 /** @brief Find symbol only in current scope (const). */
95 const SymbolEntry* lookupInCurrent(const std::string& name) const;
96 /** @brief Find symbol only in current scope (mutable). */
97 SymbolEntry* lookupMutableInCurrent(const std::string& name);
98 /** @brief Resolve symbol through current scope then ancestors. */
99 const SymbolEntry* resolve(const std::string& name) const;
100
101 /** @brief Get scope display name. */
102 const std::string& getScopeName() const;
103 /** @brief Get parent scope (null for root). */
104 std::shared_ptr<SymbolTable> getParent() const;
105 /** @brief Get child scopes. */
106 const std::vector<std::shared_ptr<SymbolTable>>& getChildren() const;
107 /** @brief Get all entries defined in this scope. */
108 const std::vector<SymbolEntry>& getEntries() const;
109
110 private:
111 std::string _scopeName;
112 std::weak_ptr<SymbolTable> _parent;
113 std::vector<std::shared_ptr<SymbolTable>> _children;
114 std::vector<SymbolEntry> _entries;

116};
