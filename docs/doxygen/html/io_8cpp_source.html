29std::tuple<std::vector<std::vector<Token>>, std::vector<std::vector<Token>>> tokenizeFile(const std::string& filename) {
30 std::tuple<std::vector<std::vector<Token>>, std::vector<std::vector<Token>>> valid_and_invalid_tokens;
31 std::ifstream file(filename);
32
33 if (!file.is_open()) {
34 throw std::runtime_error("Could not open file: " + filename);
35 }
36
37 std::string line;
38 int lineNumber = 1;
39 bool inBlockComment = false;
40
41 while (std::getline(file, line)) {
42 auto token_vectors = Token::tokenize(line, lineNumber, inBlockComment);
43 std::vector<Token> valids = std::get<0>(token_vectors);
44 std::vector<Token> invalids = std::get<1>(token_vectors);
45
46 std::get<0>(valid_and_invalid_tokens).push_back(valids);
47 std::get<1>(valid_and_invalid_tokens).push_back(invalids);
48
49 lineNumber++;
50 }
51
52 // If we reached EOF while still inside a block comment, flush it as an error
53 if (inBlockComment) {
54 std::vector<Token> pendingValid;
55 std::vector<Token> pendingInvalid;
56 Token::flushPendingBlockComment(pendingValid, pendingInvalid);
57 if (!pendingValid.empty()) {
58 std::get<0>(valid_and_invalid_tokens).push_back(pendingValid);
59 }
60 if (!pendingInvalid.empty()) {
61 std::get<1>(valid_and_invalid_tokens).push_back(pendingInvalid);
62 }
63 }
64
65 file.close();
66
67 return valid_and_invalid_tokens;
68}
