155std::tuple<std::vector<std::vector<Token>>, std::vector<std::vector<Token>>> tokenizeFile(const std::string& filename) {
156 std::tuple<std::vector<std::vector<Token>>, std::vector<std::vector<Token>>> valid_and_invalid_tokens;
157 std::ifstream file(filename);
158
159 if (!file.is_open()) {
160 throw std::runtime_error("Could not open file: " + filename);
161 }
162
163 std::string line;
164 int lineNumber = 1;
165 bool inBlockComment = false;
166
167 while (std::getline(file, line)) {
168 auto token_vectors = Token::tokenize(line, lineNumber, inBlockComment);
169 std::vector<Token> valids = std::get<0>(token_vectors);
170 std::vector<Token> invalids = std::get<1>(token_vectors);
171
172 std::get<0>(valid_and_invalid_tokens).push_back(valids);
173 std::get<1>(valid_and_invalid_tokens).push_back(invalids);
174
175 lineNumber++;
176 }
177
178 // If we reached EOF while still inside a block comment, flush it as an error
179 if (inBlockComment) {
180 std::vector<Token> pendingValid;
181 std::vector<Token> pendingInvalid;
182 Token::flushPendingBlockComment(pendingValid, pendingInvalid);
183 if (!pendingValid.empty()) {
184 std::get<0>(valid_and_invalid_tokens).push_back(pendingValid);
185 }
186 if (!pendingInvalid.empty()) {
187 std::get<1>(valid_and_invalid_tokens).push_back(pendingInvalid);
188 }
189 }
190
191 file.close();
192
193 return valid_and_invalid_tokens;
194}
